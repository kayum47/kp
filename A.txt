int main() {
    int a, b;
    char op;

    printf("Enter operator (+ - * /): ");
    scanf(" %c", &op);
    printf("Enter two numbers: ");
    scanf("%d %d", &a, &b);

    switch (op) {
        case '+': printf("Result = %d", a + b); break;
        case '-': printf("Result = %d", a - b); break;
        case '*': printf("Result = %d", a * b); break;
        case '/': printf("Result = %d", a / b); break;
        default: printf("Invalid operator");
    }
    return 0;
}

üîü Palindrome Number

#include <stdio.h>

int main() {
    int n, rev = 0, temp;
    printf("Enter number: ");
    scanf("%d", &n);

    temp = n;
    while (n > 0) {
        rev = rev * 10 + n % 10;
        n /= 10;
    }

    if (rev == temp)
        printf("Palindrome");
    else
        printf("Not Palindrome");

    return 0;
}

    printf("Fibonacci Series: ");

‚úÖ C Program: Convert Days into Years, Months, and Days

#include <stdio.h>

int main() {
    int days, years, months, remaining_days;

    printf("Enter total number of days: ");
    scanf("%d", &days);

    years = days / 365;
    days = days % 365;

    months = days / 30;
    remaining_days = days % 30;

    printf("Years = %d\n", years);
    printf("Months = %d\n", months);
    printf("Days = %d\n", remaining_days);

    return 0;
}

üß† Logic Explanation (For Viva / Exam)

1. Divide total days by 365 to get years


2. Find remaining days using modulus (%)


3. Divide remaining days by 30 to get months


4. Remaining value gives final days
